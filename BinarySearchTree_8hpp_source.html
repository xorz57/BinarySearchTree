71 std::shared_ptr<BinarySearchTreeNode> InsertHelper(std::shared_ptr<BinarySearchTreeNode> root, const Key &key, const Value &value) {
72 if (!root) return std::make_shared<BinarySearchTreeNode>(key, value);
73 if (key == root->key) {
74 root->value = value;
75 } else if (key < root->key) {
76 root->lChild = InsertHelper(root->lChild, key, value);

83 std::shared_ptr<BinarySearchTreeNode> RemoveHelper(std::shared_ptr<BinarySearchTreeNode> root, const Key &key) {
84 if (!root) return nullptr;

89 return root->lChild;

92 root->key = minRight->key;

94 root->rChild = RemoveHelper(root->rChild, minRight->key);
95 } else if (key < root->key) {
